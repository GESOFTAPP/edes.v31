function gestionRegularizacion(){
	if( eGF('tipo_regulariza') == 'T' ) {
		ePF('importe_devol',eGF('importe')); 
		ePF('cd_tipo_regu','');
		eEF('cd_tipo_regu',0);
	}else if( eGF('tipo_regulariza') == 'P' ) {
		ePF('importe_devol','');
		eEF('cd_tipo_regu',1);
	}else{
		ePF('importe_devol','');
		eEF('cd_tipo_regu',0);
	
	}
}
